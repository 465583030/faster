537  T * result;
538  size_t rSize;
539 
540  if (op == OP_Reduce){
541  T * pResult;

543

545

548 
549  // Do the rest of the reduces
550  for (int i = 1; i < (this->context->numProcs() - 1); ++i){
551  pResult = result;
552  prSize = rSize;
553  std::pair<T*,size_t> r = reduceFunc(pResult, prSize, partResult[i], partrSize[i]);

561  result = r.first;
562  rSize = r.second;
563  }

565  std::vector<T> vResult(rSize);
566  vResult.assign(result, result + rSize);
